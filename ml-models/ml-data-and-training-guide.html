# Primary: GOV.UK wheat price index (weekly, ~148 weeks) gov = pd.read_csv("govuk_price_indices_API.csv") gov_w = gov[gov["category"] == "wheat"].copy() gov_w["date"] = pd.to_datetime(gov_w["date"]) series = gov_w.groupby("date")["index"].mean().resample("W").mean().dropna() series.name = "price_index" # Optional: also load DEFRA £/t for dissertation citation defra = pd.read_csv("defra_ammg_combined.csv") wheat_gbp = defra[defra["item"] == "bread_milling_wheat"].copy() print("GOV.UK wheat weeks:", len(series)) print(series.tail()) series.to_csv("price_weekly.csv") print("Saved price_weekly.csv for Notebook D")

Cell 3 — Prepare scaler & sequence helper

Copy into Colab cell 3
TEST_WEEKS = 20 EPOCHS = 40 values = series.values.reshape(-1, 1) scaler = MinMaxScaler() scaled = scaler.fit_transform(values) def make_sequences(data, seq_len): X, y = [], [] for i in range(seq_len, len(data)): X.append(data[i-seq_len:i, 0]) y.append(data[i, 0]) return np.array(X), np.array(y) def build_lstm(seq_len): m = Sequential([ LSTM(64, return_sequences=True, input_shape=(seq_len, 1)), Dropout(0.2), LSTM(32), Dropout(0.2), Dense(1) ]) m.compile(optimizer="adam", loss="mse") return m

Cell 4 — Optimize SEQ_LEN (pick lowest validation MAPE)

Copy into Colab cell 4 — tries 8, 12, 16 week lookback
def mape(y_true, y_pred): y_true, y_pred = np.array(y_true), np.array(y_pred) return np.mean(np.abs((y_true - y_pred) / y_true)) * 100 tune_results = [] for seq_len in [8, 12, 16]: X, y = make_sequences(scaled, seq_len) split = len(X) - TEST_WEEKS X_tr, X_te = X[:split], X[split:] y_tr, y_te = y[:split], y[split:] X_tr = X_tr.reshape((X_tr.shape[0], X_tr.shape[1], 1)) X_te = X_te.reshape((X_te.shape[0], X_te.shape[1], 1)) tmp_model = build_lstm(seq_len) tmp_model.fit(X_tr, y_tr, epochs=15, batch_size=16, verbose=0, validation_split=0.1) pred = tmp_model.predict(X_te, verbose=0).flatten() y_true_tmp = scaler.inverse_transform(y_te.reshape(-1, 1)).flatten() y_pred_tmp = scaler.inverse_transform(pred.reshape(-1, 1)).flatten() val_mape = mape(y_true_tmp, y_pred_tmp) tune_results.append({"seq_len": seq_len, "mape": val_mape}) print(f"SEQ_LEN={seq_len} → MAPE {val_mape:.2f}%") best = min(tune_results, key=lambda x: x["mape"]) SEQ_LEN = best["seq_len"] print(f"\nBest SEQ_LEN: {SEQ_LEN} (MAPE {best['mape']:.2f}%)")

Cell 5 — Train final LSTM with best SEQ_LEN

Copy into Colab cell 5 — Runtime → GPU recommended
X, y = make_sequences(scaled, SEQ_LEN) split = len(X) - TEST_WEEKS X_train, X_test = X[:split], X[split:] y_train, y_test = y[:split], y[split:] X_train = X_train.reshape((X_train.shape[0], X_train.shape[1], 1)) X_test = X_test.reshape((X_test.shape[0], X_test.shape[1], 1)) model = build_lstm(SEQ_LEN) model.fit(X_train, y_train, epochs=EPOCHS, batch_size=16, verbose=1, validation_split=0.1) pred_scaled = model.predict(X_test, verbose=0).flatten() y_true = scaler.inverse_transform(y_test.reshape(-1, 1)).flatten() y_pred = scaler.inverse_transform(pred_scaled.reshape(-1, 1)).flatten() print("Final LSTM trained.")

train_series = series.iloc[:split + SEQ_LEN] test_series = series.iloc[split + SEQ_LEN:] arima = ARIMA(train_series, order=(2, 1, 2)) arima_fit = arima.fit() arima_pred = arima_fit.forecast(steps=len(test_series)) lstm_mape = mape(y_true, y_pred) arima_mape = mape(test_series.values, arima_pred.values) lstm_rmse = float(np.sqrt(mean_squared_error(y_true, y_pred))) lstm_mae = float(mean_absolute_error(y_true, y_pred)) metrics_b = { "model": "L3 LSTM price", "optimization": "SEQ_LEN grid [8,12,16]", "best_seq_len": int(SEQ_LEN), "tune_results": tune_results, "baseline": "ARIMA(2,1,2)", "lstm_mape_pct": round(lstm_mape, 2), "arima_mape_pct": round(arima_mape, 2), "lstm_rmse": round(lstm_rmse, 4), "lstm_mae": round(lstm_mae, 4), "lstm_beats_arima": bool(lstm_mape < arima_mape), "nfr4_target_pct": 15, "nfr4_pass": bool(lstm_mape < 15), } print(f"LSTM MAPE: {metrics_b['lstm_mape_pct']}%") print(f"ARIMA MAPE: {metrics_b['arima_mape_pct']}%") print(f"LSTM RMSE: {metrics_b['lstm_rmse']}") print(f"LSTM MAE: {metrics_b['lstm_mae']}") print(f"LSTM beats ARIMA? {metrics_b['lstm_beats_arima']}") print(f"NFR4 pass? {'YES' if metrics_b['nfr4_pass'] else 'NO'}") with open("metrics_b.json", "w") as f: json.dump(metrics_b, f, indent=2)

Cell 3 — Google Trends (with retry) OR offline fallback

Copy into Colab cell 3 — if 429 persists, fallback runs automatically
from pytrends.exceptions import TooManyRequestsError keywords = ["wheat", "potato", "barley"] # one at a time reduces 429 errors TRENDS_PAUSE = 30 # seconds between keywords / retries TRENDS_RETRIES = 4 def fetch_trends_one_by_one(keywords, timeframe="today 3-y", geo="GB"): pytrends = TrendReq(hl="en-GB", tz=0, retries=2, backoff_factor=1.0) parts = [] for kw in keywords: ok = False for attempt in range(TRENDS_RETRIES): try: wait = TRENDS_PAUSE * (attempt + 1) if attempt > 0: print(f"Retry {kw} in {wait}s...") time.sleep(wait) else: time.sleep(10) pytrends.build_payload([kw], timeframe=timeframe, geo=geo) df = pytrends.interest_over_time() if df is None or df.empty: raise ValueError("empty response") if "isPartial" in df.columns: df = df.drop(columns=["isPartial"]) parts.append(df[[kw]]) print(f"✓ Downloaded Trends: {kw}") time.sleep(TRENDS_PAUSE) ok = True break except (TooManyRequestsError, Exception) as e: print(f"✗ {kw} attempt {attempt+1}/{TRENDS_RETRIES}: {type(e).__name__}") if not ok: print(f"Could not fetch {kw} from Google Trends") if not parts: return None out = pd.concat(parts, axis=1) return out.resample("W").mean().dropna() def offline_sentiment_proxy(keywords, price_index): """Fallback when Google returns 429. Twitter has no dates — proxy only.""" twitter = pd.read_csv("twitter_data.csv") n = len(price_index) proxy = pd.DataFrame(index=price_index.index) for kw in keywords: mask = twitter["clean_text"].str.contains(kw, case=False, na=False) sub = twitter.loc[mask] base = (sub["category"].mean() + 1.0) if len(sub) else 0.0 # spread mention signal across weeks (document limitation in Ch.7) proxy[kw] = base + np.linspace(0, len(sub) / max(n, 1), n) * 0.01 return proxy trends_source = "google_trends" trends = fetch_trends_one_by_one(keywords) if trends is None or trends.empty: trends_source = "offline_sentiment_proxy" prices_tmp = pd.read_csv("price_weekly.csv", index_col=0, parse_dates=True) trends = offline_sentiment_proxy(keywords, prices_tmp) print("\n⚠ Trends blocked (429). Using offline sentiment proxy.") print("Write in Ch.7: live Trends failed; offline proxy used for Panel D.") print(trends.tail()) print("Source:", trends_source)
